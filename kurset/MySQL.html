Modulet e Kursit

1. Hyrje në MySQL

MySQL është një sistem për menaxhimin e bazave të të dhënave. Në këtë modul mësoni çfarë është SQL dhe si funksionon.

-- Shfaq të gjitha databazat
SHOW DATABASES;

2. Krijimi i databazave dhe tabelave

Si të krijoni databazë dhe tabela brenda saj.

-- Krijo një databazë
CREATE DATABASE shkolla;

-- Përdor databazën
USE shkolla;

-- Krijo një tabelë
CREATE TABLE studentet (
    id INT PRIMARY KEY AUTO_INCREMENT,
    emri VARCHAR(50),
    mbiemri VARCHAR(50),
    mosha INT
);

3. INSERT - Shtimi i të dhënave

Si të fusni të dhëna në tabelë.

INSERT INTO studentet (emri, mbiemri, mosha) 
VALUES ('Arlinda', 'Behrami', 19);

4. SELECT - Marrja e të dhënave

Mësoni si të merrni të dhëna nga një tabelë.

-- Merr të gjitha rreshtat
SELECT * FROM studentet;

-- Merr vetëm emrin dhe moshën
SELECT emri, mosha FROM studentet;

5. UPDATE - Përditësimi i të dhënave

Si të ndryshoni të dhënat ekzistuese.

UPDATE studentet
SET mosha = 20
WHERE id = 1;

6. DELETE - Fshirja e të dhënave

Si të fshini rreshta nga tabela.

DELETE FROM studentet
WHERE id = 1;

7. WHERE, AND, OR (Filtrimi)

Filtro të dhënat sipas kushteve.

SELECT * FROM studentet
WHERE mosha > 18 AND mbiemri = 'Behrami';

8. ORDER BY - Renditja

Si të renditni të dhënat sipas një kolone.

SELECT * FROM studentet
ORDER BY mosha DESC;

9. Funksionet (COUNT, MAX, MIN, AVG)

Mësoni funksionet kryesore në MySQL.

SELECT COUNT(*) FROM studentet;
SELECT AVG(mosha) FROM studentet;

10. JOIN - Bashkimi i tabelave

Si të bashkoni dy tabela me të dhënat që lidhen mes vete.

SELECT s.emri, k.emri_kursit
FROM studentet s
JOIN kurset k ON s.id = k.student_id;
